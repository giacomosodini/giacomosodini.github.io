\newcommand{\loaditemsfromfile}[2]{%
\IfFileExists{#1}{%
  \begingroup
  % Le intestazioni "### Nome sezione" contengono '#', che va reso innocuo per la lettura del file
  \catcode`\#=12
  \catcode`\%=12
  \catcode`\&=12
  \catcode`\_=12
  \catcode`\~=12
  \catcode`\^=12
  \catcode`\$=12
  \readingtafalse
  \openin\myread=#1
  \resumeItemListStart
    \loop\unless\ifeof\myread%
    \read\myread to \localvariable
    \ifeof\myread
    \else
        % Attiva la sottosezione
        \IfSubStr{\localvariable}{#2}{\readingtatrue}{}%
        % Disattiva alla fine della sottosezione
        \IfSubStr{\localvariable}{---}{\readingtafalse}{}%
        \ifreadingta
    	\IfBeginWith{\localvariable}{- **}{
	\StrBetween[1,1]{\localvariable}{- **}{** -}[\thedate]%
	\StrBehind{\localvariable}{** - }[\therest]%
	\IfBeginWith{\therest}{*"}{
	  \StrBetween[1,1]{\localvariable}{- *"}{"* -}[\thetitle]%
	  \StrBehind{\localvariable}{"* - }[\thelocation]
	  \item \textbf{\thedate} \emph{\thetitle}, \thelocation
	}{
	  \item \textbf{\thedate} \therest
	}%

 }
    {}%
        \fi
    \fi
    \repeat
  \resumeItemListEnd % close the environment
  \closein\myread%  \close the file
  \endgroup
}{}%
}

% Converte i link in stile Markdown "[testo](url)" in \href{url}{testo},
% cercando ogni occorrenza in successione (gestisce più link sulla stessa riga).
\newcommand{\ConvertMDLinks}[1]{%
  \IfSubStr{#1}{[}{%
    \StrBefore{#1}{[}[\mdprefix]%
    \StrBetween{#1}{[}{]}[\mdtext]%
    \StrBehind{#1}{]}[\mdrest]%
    \StrBetween{\mdrest}{(}{)}[\mdurl]%
    \StrBehind{\mdrest}{)}[\mdsuffix]%
    \mdprefix\href{\mdurl}{\mdtext}\ConvertMDLinks{\mdsuffix}%
  }{%
    #1%
  }%
}

\newcommand{\loaditemta}[2]{%
  \IfFileExists{#1}{%
    \begingroup
    % Gli URL letti da file possono contenere questi caratteri speciali di TeX;
    % li rendiamo "normali" solo per la lettura, altrimenti verrebbero troncati o alterati.
    \catcode`\%=12
    \catcode`\&=12
    \catcode`\#=12
    \catcode`\_=12
    \catcode`\~=12
    \catcode`\^=12
    \catcode`\$=12
    \openin\myread=#1
    \readingtafalse
    \initemizefalse
    \loop\unless\ifeof\myread
      \read\myread to \localline
      \ifeof\myread\else

        % Attiva la sezione "Teaching Assistant"
        \IfSubStr{\expandafter\detokenize\expandafter{\localline}}{\expandafter\detokenize\expandafter{#2}}{\readingtatrue}{}%

        % Disattiva alla fine della sezione
        \IfSubStr{\expandafter\detokenize\expandafter{\localline}}{---}{%
          \readingtafalse
        }{}%
        \ifreadingta
          % Riga con nome università
          \IfBeginWith{\expandafter\detokenize\expandafter{\localline}}{- **}{%
          \StrBetween[1,1]{\expandafter\detokenize\expandafter{\localline}}{- **}{**:}[\theplace]
          \global\let\theplace\theplace % StrBetween assegna localmente: rendiamolo globale prima di chiudere la lista, altrimenti si perde il valore
          \ifinitemize\resumeItemListEnd\initemizefalse\fi
            \vspace{-0.4em}\item \textbf{\theplace} \vspace{1.1em}
          }{%
            % Riga con corso
            \IfBeginWith{\expandafter\detokenize\expandafter{\localline}}{- }{%
              \StrBehind{\expandafter\detokenize\expandafter{\localline}}{- }[\course]
              \global\let\course\course
              \ifx\course\empty\else
                \ifinitemize\else\vspace{-2.25em}\resumeItemListStart\initemizetrue\fi
                \item \ConvertMDLinks{\course}
              \fi
            }{}%
          }%
        \fi
      \fi
    \repeat
    \ifinitemize\resumeItemListEnd\initemizefalse\fi
    \closein\myread
    \endgroup
  }{%
    File not found%
  }%
}
